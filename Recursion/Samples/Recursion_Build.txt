//Build a LLL - first iteratively
//Add at the end

void build(node * & head, char * array)
{

  //the list is new....head is NULL to start
  if (head == NULL) //if(!head)
  {
     head = new node;
     head->name = new char[strlen(array)+1];
     strcpy(head->name, array);
     head->next = NULL;
  }
  else
  {
     node * current = head;
     while (current->next)//traverse TO last node
        current= current->next;
     current->next = new node; //connect up!
     current= current->next; //travel TO new last 
     current->name = new char[strlen(array)+1];
     strcpy(current->name, array);
     current->next = NULL;
   }
}
     
//Build a LLL - first recursively
//Add at the end

void build(node * & head, char * array)
{

  //the list is new....head is NULL to start
  if (head == NULL) //if(!head)
  {
     head = new node;
     head->name = new char[strlen(array)+1];
     strcpy(head->name, array);
     head->next = NULL;
  }
  else
     build(head->next, array);
}
